    <script>
        async function generarVoz() {
            const texto = document.getElementById('textoInput').value;
            const loader = document.getElementById('loader');
            const player = document.getElementById('player');
            
            // TU URL CONFIGURADA
            const url_backend = "https://creatorforge-ai-2.onrender.com"; 
            
            if (!texto.trim()) {
                alert("Por favor, escribe algo para Alonso.");
                return;
            }

            loader.style.display = "block";

            try {
                // Generamos la URL del audio
                const finalUrl = `${url_backend}/tts?text=${encodeURIComponent(texto)}&voice=es-MX-AlonsoNeural`;
                
                player.src = finalUrl;
                
                // Intentar reproducir
                player.play().catch(e => {
                    console.error("Error al reproducir:", e);
                    alert("Haz clic en la página primero para permitir el audio.");
                });
                
                player.onplaying = () => {
                    loader.style.display = "none";
                };

                player.onerror = () => {
                    loader.style.display = "none";
                    alert("Error: El servidor está despertando, intenta de nuevo en 20 segundos.");
                };

            } catch (error) {
                console.error(error);
                loader.style.display = "none";
            }
        }
    </script>
